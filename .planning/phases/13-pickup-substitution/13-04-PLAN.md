    **Upfront Pickup Processing (triggered when user sets pickup file path):**
    ```csharp
    private async Task HandlePickupFileSet()
    {
        if (string.IsNullOrWhiteSpace(_pickupSessionPath)) return;
        _isProcessing = true;
        _processingStatus = "Preparing chapters...";
        _processingProgress = 0;
        StateHasChanged();

        try
        {
            // 1. Gather flagged sentences from all CRX chapters
            var chapterTargets = new List<(string Stem, string Name, IReadOnlyList<HydratedSentence> Flagged)>();
            foreach (var ch in _chaptersWithCrx)
            {
                Workspace.SelectChapter(ch.ChapterName);
                if (Workspace.TryGetHydratedTranscript(ch.ChapterName, out var hydrated) && hydrated != null)
                {
                    var flagged = LoadFlaggedSentencesForChapter(ch.ChapterName, hydrated);
                    if (flagged.Count > 0)
                        chapterTargets.Add((ch.Stem, ch.ChapterName, flagged));
                }
            }

            // Restore to current chapter
            if (_currentChapterIndex >= 0)
                Workspace.SelectChapter(_chaptersWithCrx[_currentChapterIndex].ChapterName);

            // 2. Run cross-chapter import with progress
            var progress = new Progress<(string Status, double Progress)>(p =>
            {
                _processingStatus = p.Status;
                _processingProgress = p.Progress;
                InvokeAsync(StateHasChanged);
            });

            _matchesByChapter = await PolishService.ImportPickupsCrossChapterAsync(
                _pickupSessionPath.Trim(), chapterTargets, progress, CancellationToken.None);

            _processingStatus = $"Found {_matchesByChapter.Values.Sum(m => m.Count)} matches across {_matchesByChapter.Count} chapters.";
        }
        catch (Exception ex) { _processingStatus = $"Processing failed: {ex.Message}"; }
        finally { _isProcessing = false; StateHasChanged(); }
    }
    ```

    **Stage Action (from PickupBox OnStage):**
    ```csharp
    private void HandleStage(PickupMatch match)
    {
        var currentChapter = _chaptersWithCrx[_currentChapterIndex];
        var sentence = _flaggedSentences?.FirstOrDefault(s => s.Id == match.SentenceId);
        if (sentence?.Timing == null) return;

        PolishService.StageReplacement(
            currentChapter.Stem, match, _pickupSessionPath!,
            sentence.Timing.StartSec, sentence.Timing.EndSec);
        RefreshCurrentChapterState();
    }
    ```
