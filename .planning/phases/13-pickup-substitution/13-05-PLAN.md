    ```csharp
    private async Task CheckCompletionAndAdvance()
    {
        var matchCount = GetMatchesForCurrentChapter().Count;
        var committedCount = GetCommittedItems().Count;

        if (matchCount > 0 && committedCount >= matchCount)
        {
            // Mark current chapter as complete
            _chaptersWithCrx[_currentChapterIndex] = _chaptersWithCrx[_currentChapterIndex]
                with { AllCommitted = true };

            // Auto-advance to next incomplete chapter
            var nextIncomplete = FindNextIncompleteChapter();
            if (nextIncomplete >= 0)
            {
                await Task.Delay(500); // Brief pause for visual feedback
                await NavigateToChapter(nextIncomplete);
            }
        }
    }

    private int FindNextIncompleteChapter()
    {
        for (int i = _currentChapterIndex + 1; i < _chaptersWithCrx.Count; i++)
        {
            if (!_chaptersWithCrx[i].AllCommitted) return i;
        }
        // Wrap around
        for (int i = 0; i < _currentChapterIndex; i++)
        {
            if (!_chaptersWithCrx[i].AllCommitted) return i;
        }
        return -1; // All complete
    }
    ```
