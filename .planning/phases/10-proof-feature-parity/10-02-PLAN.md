**ChapterCard.razor** - Clickable chapter summary:
```razor
<BitCard Class="chapter-card @GetStatusClass()" @onclick="HandleClick" Style="cursor: pointer;">
    <BitStack Horizontal="false" Gap="0.5rem">
        <BitText Typography="BitTypography.Subtitle2">@Chapter.Name</BitText>
        <div class="chapter-metrics">
            <span class="metric">
                <span class="label">Sentences:</span>
                <span class="value">@Chapter.Metrics.SentenceCount (@Chapter.Metrics.SentenceFlagged flagged)</span>
            </span>
            <span class="metric">
                <span class="label">Avg WER:</span>
                <span class="value @GetWerClass()">@Chapter.Metrics.SentenceAvgWer</span>
            </span>
        </div>
    </BitStack>
</BitCard>

@code {
    [Parameter] public ChapterInfo Chapter { get; set; } = default!;
    [Parameter] public EventCallback OnClick { get; set; }

    private async Task HandleClick() => await OnClick.InvokeAsync();

    private string GetStatusClass()
    {
        // Future: add "reviewed" class when review status is implemented
        return "";
    }

    private string GetWerClass()
    {
        // Parse WER percentage and classify
        if (double.TryParse(Chapter.Metrics.SentenceAvgWer.TrimEnd('%'), out var wer))
        {
            if (wer >= 10) return "wer-high";
            if (wer >= 5) return "wer-medium";
        }
        return "wer-low";
    }
}
```
