**ErrorCard.razor** - Compact error summary for quick grid:
```razor
<BitCard Class="error-card" @onclick="HandleClick" Style="cursor: pointer;">
    <BitStack Horizontal="false" Gap="0.25rem">
        <BitStack Horizontal="true" HorizontalAlign="BitAlignment.SpaceBetween">
            <BitText Typography="BitTypography.Caption1">#@Sentence.Id</BitText>
            <BitBadge Color="@GetStatusColor()">@Sentence.Status</BitBadge>
        </BitStack>
        <div class="error-metrics">
            <span class="metric">WER: <span class="@GetWerClass()">@Sentence.Wer</span></span>
            <span class="metric">CER: @Sentence.Cer</span>
        </div>
        <BitText Typography="BitTypography.Caption2" Color="BitColor.SecondaryForeground"
                 Style="white-space: nowrap; overflow: hidden; text-overflow: ellipsis;">
            @Sentence.Excerpt
        </BitText>
    </BitStack>
</BitCard>

@code {
    [Parameter] public SentenceReport Sentence { get; set; } = default!;
    [Parameter] public EventCallback OnClick { get; set; }

    private async Task HandleClick() => await OnClick.InvokeAsync();

    private BitColor GetStatusColor() => Sentence.Status switch
    {
        "unreliable" => BitColor.Error,
        "attention" => BitColor.Warning,
        _ => BitColor.Success
    };

    private string GetWerClass()
    {
        if (double.TryParse(Sentence.Wer.TrimEnd('%'), out var wer))
        {
            if (wer >= 50) return "wer-high";
            if (wer >= 10) return "wer-medium";
        }
        return "wer-low";
    }
}
```
