    public ErrorPatternsResult AggregatePatterns(ISet<string>? ignoredKeys = null)
    {
        var patterns = new Dictionary<string, (ErrorPattern Pattern, List<PatternExample> Examples)>();
        ignoredKeys ??= new HashSet<string>();

        // Iterate all available chapters
        foreach (var chapterTitle in _workspace.AvailableChapters)
        {
            // Open chapter and get hydrate
            // Extract patterns from each sentence's diff
        }

        // Build final result with top 3 examples each
        return new ErrorPatternsResult(
            patterns.Values
                .Select(p => p.Pattern with { Examples = p.Examples.Take(3).ToList() })
                .OrderByDescending(p => p.Count)
                .ToList(),
            ignoredKeys.ToList()
        );
    }

    /// <summary>
    /// Extract patterns from a single sentence's diff operations.
    /// </summary>
    private IEnumerable<(string Type, string Book, string Script)> ExtractPatterns(HydratedDiff? diff)
    {
        if (diff?.Ops == null) yield break;

        var ops = diff.Ops.ToList();
        for (int i = 0; i < ops.Count; i++)
        {
            var op = ops[i];

            if (op.Operation == "delete")
            {
                // Check if next op is insert (= substitution)
                if (i + 1 < ops.Count && ops[i + 1].Operation == "insert")
                {
                    var next = ops[i + 1];
                    yield return ("sub", string.Join(" ", op.Tokens), string.Join(" ", next.Tokens));
                    i++; // Skip next op
                }
                else
                {
                    yield return ("del", string.Join(" ", op.Tokens), "");
                }
            }
            else if (op.Operation == "insert")
            {
                // Standalone insert (not part of substitution)
                yield return ("ins", "", string.Join(" ", op.Tokens));
            }
            // "equal" ops are ignored
        }
    }

    /// <summary>
    /// Build unique pattern key.
    /// </summary>
    public static string BuildKey(string type, string book, string script)
        => $"{type}|{book}|{script}";
}

    <BitTextField @bind-Value="_filterText"

                  OnInput="HandleFilterInput"

    @if (_loading)
    {
        <BitProgress Circular="true" />
    }
    else if (_patterns != null)
    {
        <div class="patterns-grid">
            @foreach (var pattern in FilteredPatterns)
            {
                <PatternCard Pattern="pattern"
                             OnToggleIgnore="() => ToggleIgnore(pattern)"
                             OnExampleClick="NavigateToExample" />
            }
        </div>

        @if (!FilteredPatterns.Any())
        {
            <BitText Color="BitColor.SecondaryForeground" Style="font-style: italic;">
                No patterns match the filter.
            </BitText>
        }
    }
    else
    {
        <BitMessage Color="BitColor.Warning">
            No pattern data available. Set a working directory first.
        </BitMessage>
    }

@code {
    private ErrorPatternsResult? _patterns;
    private string _filterText = "";
    private bool _loading = true;

    protected override async Task OnInitializedAsync()
    {
        await LoadPatterns();
    }

    private async Task LoadPatterns()
    {
        _loading = true;
        try
        {
            _patterns = await Http.GetFromJsonAsync<ErrorPatternsResult>("/api/proof/errors/aggregate");
        }
        catch { }
        _loading = false;
    }

    private IEnumerable<ErrorPattern> FilteredPatterns =>
        _patterns?.Patterns.Where(p =>
            string.IsNullOrEmpty(_filterText) ||
            p.Book.Contains(_filterText, StringComparison.OrdinalIgnoreCase) ||
            p.Script.Contains(_filterText, StringComparison.OrdinalIgnoreCase)
        ) ?? Enumerable.Empty<ErrorPattern>();

    private void HandleFilterInput(ChangeEventArgs e)
    {
        _filterText = e.Value?.ToString() ?? "";
        StateHasChanged();
    }

    }

    private void NavigateToExample(PatternExample example)
    {
        Navigation.NavigateTo($"/proof/{Uri.EscapeDataString(example.Chapter)}?sentenceId={example.SentenceId}");
